-- CREAZIONE TABELLE

CREATE TABLE Utente(
    Email VARCHAR(50) PRIMARY KEY,
    NomeUtente VARCHAR(50) NOT NULL,
    CognomeUtente VARCHAR(50) NOT NULL,
    HashPWD VARCHAR(100) NOT NULL,
    Citta VARCHAR(30) NOT NULL,
    Nazione VARCHAR(30) NOT NULL,
    Ristoratore BOOLEAN NOT NULL
);

CREATE TABLE TipiCucina(
    IDTipoCucina SERIAL PRIMARY KEY,
    TipoCucina VARCHAR(30) UNIQUE NOT NULL
);

CREATE TABLE Ristorante(
    IDRistorante SERIAL PRIMARY KEY,
    NomeRistorante VARCHAR(50) NOT NULL,
    Email VARCHAR(50) NOT NULL,
    Citta VARCHAR(50) NOT NULL,
    Nazione VARCHAR(50) NOT NULL,
    Via VARCHAR(50) NOT NULL,
    NumeroCivico INT NOT NULL,
    FasciaPrezzo INT NOT NULL,
    Delivery BOOLEAN NOT NULL,
    PrenotazioneOnline BOOLEAN NOT NULL,
    IDTipoCucina INT NOT NULL,
    FOREIGN KEY (Email)
        REFERENCES Utente(Email) ON DELETE CASCADE ON UPDATE CASCADE,
    FOREIGN KEY (IDTipoCucina)
         REFERENCES TipiCucina(IDTipoCucina) ON DELETE SET NULL ON UPDATE CASCADE
);

CREATE TABLE Preferiti(
    IDRistorante INT,
    Email VARCHAR(50),
    PRIMARY KEY (IDRistorante, Email)
    FOREIGN KEY (IDRistorante)
        REFERENCES Ristorante(IDRistorante)  ON DELETE CASCADE ON UPDATE CASCADE,
    FOREIGN KEY (Email)
        REFERENCES Utente(Email) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE Recensioni(
    Email VARCHAR(50),
    IDRistorante INT,
    Valutazione INT NOT NULL,
    Recensione VARCHAR(750),
    Risposta VARCHAR(750),
    PRIMARY KEY (Email, IDRistorante)
    FOREIGN KEY (Email)
        REFERENCES Utente(Email) ON DELETE CASCADE ON UPDATE CASCADE
    FOREIGN KEY (IDRistorante)
        REFERENCES Ristorante(IDRistorante) ON DELETE CASCADE ON UPDATE CASCADE
);
